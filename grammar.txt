// Grammar for the double u programming language

            program : statementList EOF
    
      statementList : statement
                    | statement statementList

          statement : expr SEMI
                    | loop
                    | if
                    | funcDef
                    | println

               expr : firstClass
                    | firstClass op expr

         firstClass : IDENTIFIER
                    | STRING
                    | INTEGER
                    | boolean
                    | lambda
                    | funcDef
                    | length
                    | OPAREN expr CPAREN
