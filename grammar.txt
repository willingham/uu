// Grammar for the double u programming language

            program : expressionList
    
     expressionList : expression
                    | expression expressionList

         expression : expr SEMI
                    | loop
                    | if
                    | funcDef
                    | print

               expr : primary
                    | primary operator expr

         optParamList : *empty*
                    | paramList

            argList : expr
                    | expr COMMA argList

            primary : literal
                    | funcDef
                    | OP expr CP
                    | lambda
                    | ID OP optParamList CP
                    | operator PRIMARY

            operator : MINUS
                     | PLUS
                     | NOT
                     | GT
                     | LT
                     | GTE
                     | LTE
